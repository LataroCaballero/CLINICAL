  <action>
    Create `frontend/src/hooks/useCerrarLote.ts`:

    ```typescript
    import { useMutation, useQueryClient } from '@tanstack/react-query';
    import { useRouter } from 'next/navigation';
    import { api } from '@/lib/api';
    import { toast } from 'sonner';

    interface CerrarLoteParams {
      profesionalId: string;
      obraSocialId: string;
      practicaIds: string[];
    }

    export function useCerrarLote() {
      const queryClient = useQueryClient();
      const router = useRouter();

      return useMutation({
        mutationFn: async ({ profesionalId, obraSocialId, practicaIds }: CerrarLoteParams) => {
          const periodo = new Date().toISOString().slice(0, 7); // YYYY-MM
          const { data } = await api.post('/finanzas/liquidaciones/crear-lote', {
            profesionalId,
            obraSocialId,
            periodo,
            practicaIds,
          });
          return data;
        },
        onSuccess: (_, variables) => {
          queryClient.invalidateQueries({
            queryKey: ['finanzas', 'practicas-pendientes-agrupadas', variables.profesionalId],
          });
          queryClient.invalidateQueries({
            queryKey: ['finanzas', 'practicas-pendientes-por-os', variables.profesionalId, variables.obraSocialId],
          });
          toast.success('Lote cerrado correctamente');
          router.push('/dashboard/facturador');
        },
        onError: () => {
          toast.error('Error al cerrar el lote');
        },
      });
    }
    ```

    Note: `periodo` derived from `new Date().toISOString().slice(0, 7)` — same pattern as `mesActual` in the facturador page. This matches the YYYY-MM format required by `CreateLoteDto`.

    ```typescript
    const params = useParams();

    const osNombre = searchParams.get('nombre') ?? 'Obra Social';
    ```

    **Live total derivation:**
    ```typescript
    const derivedTotal = (practicas ?? []).reduce((sum, p) => {
      return sum + (overrides.get(p.id) ?? p.montoPagado ?? p.monto);
    }, 0);
    ```

    **EditableCell inline component (define inside the page file or as a local function):**
    - Props: `practica: PracticaPendientePorOS`, `currentOverride: number | undefined`, `onSuccess: (practicaId: string, newMonto: number) => void`
    - Local state: `localMonto: string` initialized to `''` (empty = shows placeholder)
    - `placeholder` attribute on Input = `formatMoney(practica.montoPagado ?? practica.monto)`
    - `onBlur` handler:
      - Parse `parseFloat(localMonto)`
      - If `isNaN(parsed)` or localMonto is empty string: no-op (no PATCH)
      - If `parsed === (practica.montoPagado ?? practica.monto)`: no-op (unchanged)
      - Else: call `actualizarMonto.mutate({ practicaId: practica.id, montoPagado: parsed }, { onSuccess: () => onSuccess(practica.id, parsed), onError: () => { setLocalMonto(''); toast.error('Error al guardar el monto'); } })`
    - Input type="number", min="0", step="1"
    - No visual indicator when value differs (per CONTEXT.md: no indicator, value change is sufficient)

    **Table columns:** Paciente | Código | Descripción | Fecha | Autorizado | Pagado
    - Paciente: `practica.paciente?.nombreCompleto ?? '—'`
    - Código: `practica.codigo`
    - Descripción: `practica.descripcion`
    - Fecha: `new Date(practica.fecha).toLocaleDateString('es-AR')`
    - Autorizado: `formatMoney(practica.monto)` (right-aligned)
    - Pagado: `<EditableCell>` (right-aligned)

    **Loading state:** While `isLoading`, render skeleton rows (3 rows of gray pulse divs). Claude's discretion on exact design.

    **Empty state:** If `practicas?.length === 0`, show centered message "No hay prácticas pendientes para esta obra social."

    **Header:**
    ```
    [← Volver]  {osNombre}  [Total: {formatMoney(derivedTotal)}]  [Cerrar Lote (disabled if 0 prácticas)]
    ```
    - Back link: `<Link href="/dashboard/facturador">← Volver</Link>`
    - "Cerrar Lote" button: `disabled={!practicas || practicas.length === 0 || cerrarLote.isPending}`

    **CerrarLoteModal (inline Dialog):**
    - Controlled with `open={isModalOpen}` + `onOpenChange={setIsModalOpen}`
    - Trigger: the "Cerrar Lote" button sets `isModalOpen = true` (do NOT use DialogTrigger — button is in header, modal is separate)
    - Modal content: "¿Cerrar lote?" heading, then "Se marcarán {practicas?.length ?? 0} prácticas como PAGADO por un total de {formatMoney(derivedTotal)}."
    - Cancel button: closes modal
    - Confirm button "Confirmar cierre": calls `cerrarLote.mutate({ profesionalId: selectedProfessionalId!, obraSocialId, practicaIds: practicas!.map(p => p.id) })`, disabled while `cerrarLote.isPending`

    - DO NOT invalidate practicas query on each PATCH — use local `overrides` map for live total
    - Import `useRouter` from `next/navigation` NOT `next/router`
    - Do NOT use `window.location.href` for redirect — use `router.push()`
