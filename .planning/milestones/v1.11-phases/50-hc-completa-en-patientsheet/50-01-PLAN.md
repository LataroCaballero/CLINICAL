<objective>
Hacer que las entradas de Historia Clínica estructuradas (`primera_vez`) dentro del PatientSheet rendericen su contenido completo con chips de color (zona / diagnósticos / tratamientos) + observaciones + comentario, con paridad visual frente a `HistorialClinicoPanel` (LiveTurno) y `TurnoHCModal` (agenda). </objective>

```typescript
interface ZonaContenido {           // v1.9+ shape (Array.isArray(c.zonas))
  zonaId?: string;
  zona: string;
  diagnosticos: string[];
  otroTexto?: string;               // observación (solo en detalle, bloque etiquetado)
  tratamientos: Array<{ nombre: string; tratamientoId?: string; precio: number }>;
}

interface ContenidoPrimeraVez {
  tipo: "primera_vez";
  comentario?: string;
  zonas?: ZonaContenido[];                                              // shape nuevo
  diagnostico?: { zonas?: string[]; subzonas?: string[]; otroTexto?: string }; // legacy
  tratamientos?: { nombre: string; precio?: number }[];                // legacy
  presupuestoTotal?: number;        // solo detalle
  presupuestoId?: string | null;
}

interface ContenidoLibre { tipo?: "libre" | string; texto?: string; }
type ContenidoEntrada = ContenidoPrimeraVez | ContenidoLibre | Record<string, unknown> | null;
```
