    ```typescript
    import { IsArray, IsString, IsNumber, Min, ValidateNested } from 'class-validator';
    import { Type } from 'class-transformer';

    export class InsumoItemDto {
      @IsString()
      productoId: string;

      @IsNumber()
      @Min(0.001)
      cantidad: number;
    }

    export class SetInsumosTratamientoDto {
      @IsArray()
      @ValidateNested({ each: true })
      @Type(() => InsumoItemDto)
      insumos: InsumoItemDto[];
    }
    ```

    **2. In `tratamientos.service.ts`, add two new methods:**

    Add `setInsumos(id: string, profesionalId: string, insumos: InsumoItemDto[])`:
    - First call `this.findById(id, profesionalId)` to verify ownership
    - Run a Prisma `$transaction` that:
      1. Deletes all existing `TratamientoInsumo` where `tratamientoId === id`
      2. Creates new `TratamientoInsumo` records for each item in `insumos` array
    - Return the updated tratamiento: `this.prisma.tratamiento.findUnique({ where: { id }, include: { insumos: { include: { producto: { select: { id: true, nombre: true, costoBase: true, unidadMedida: true } } } } } })`

    Add `recalcularPrecioBase(id: string, profesionalId: string)`:
    - Call `this.findById(id, profesionalId)` for ownership check
    - Query `this.prisma.tratamientoInsumo.findMany({ where: { tratamientoId: id }, include: { producto: { select: { costoBase: true } } } })`
    - Calculate: `const precioBase = insumos.reduce((sum, item) => sum + Number(item.producto.costoBase ?? 0) * Number(item.cantidad), 0)`
    - Update: `this.prisma.tratamiento.update({ where: { id }, data: { precioBase }, include: { insumos: { include: { producto: { select: { id: true, nombre: true, costoBase: true, unidadMedida: true } } } } } })`
    - Return the updated tratamiento

    **3. Also update `findAllByProfesional` method** to include insumos in its query:
    Add `include: { insumos: { include: { producto: { select: { id: true, nombre: true, costoBase: true, unidadMedida: true } } } } }` to the existing `findMany` call.

    **Route 1: `PUT /tratamientos/:id/insumos`**
    ```typescript
    @Put(':id/insumos')
    @UseGuards(JwtAuthGuard)
    async setInsumos(
      @Param('id') id: string,
      @Body() dto: SetInsumosTratamientoDto,
      @Request() req: any,
      @Query('profesionalId') profesionalId?: string,
    ) {
      const profId = await this.getProfesionalId(req.user, profesionalId);
      return this.tratamientosService.setInsumos(id, profId, dto.insumos);
    }
    ```
