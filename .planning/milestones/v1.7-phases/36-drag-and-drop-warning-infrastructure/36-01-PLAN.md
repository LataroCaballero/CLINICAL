    export function getEtapaWarning(
      patient: KanbanPatient,
      targetEtapa: EtapaCRM
    ): string | null {
      if (targetEtapa === "PRESUPUESTO_ENVIADO" && patient.presupuesto === null) {
        return "No hay presupuesto enviado a este paciente";

      return null;
