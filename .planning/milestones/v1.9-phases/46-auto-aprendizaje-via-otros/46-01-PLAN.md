```typescript
export interface SnapshotItem { id: string; nombre: string; activo: boolean }
export interface SnapshotZona extends SnapshotItem {
  diagnosticos: SnapshotItem[];
  tratamientos: SnapshotItem[];
}
export interface ZonaAprendizajeInput {
  zona: string;
  diagnosticos: string[];
  tratamientos: { nombre: string }[];
}
export interface AccionesAprendizaje {
  zonasACrear: string[];                                        // nombres formateados, en orden de aparición
  zonasAReactivar: string[];                                    // ZonaHC ids
  diagnosticosACrear: { zonaNombre: string; nombre: string }[]; // zonaNombre = nombre exacto de zona existente o nombre formateado de zona nueva
  diagnosticosAReactivar: string[];                             // DiagnosticoHC ids
  tratamientosACrear: { zonaNombre: string; nombre: string }[];
  tratamientosAReactivar: string[];                             // TratamientoHC ids
}

export function formatearNombreAprendido(raw: string): string;
export function detectarAprendizaje(
  catalogo: SnapshotZona[],
  zonas: ZonaAprendizajeInput[],
): AccionesAprendizaje;
```

Casos de test (mínimo):

formatearNombreAprendido:
- "flacidez abdominal " → "Flacidez abdominal" (trim + primera mayúscula, resto intacto)
- "BOTOX Cuello" → "BOTOX Cuello" (solo primera letra: si ya es mayúscula no cambia nada más)
- "  ácido nuevo" → "Ácido nuevo" (funciona con acentos)

detectarAprendizaje — zonas (APR-01):
- Zona "cuello" no existente en catálogo → zonasACrear: ["Cuello"]
- Zona "ABDOMEN" con "Abdomen" activo en catálogo (match insensible a mayúsculas) → sin acciones
- Zona "Cuello" que matchea ZonaHC inactiva → zonasAReactivar: [id], NO en zonasACrear
- Zona "otros" / "Otros" → nunca se aprende (es el ítem sistema)
- Zona "" o "   " → sin acciones
- Dos zonas del input que normalizan igual ("cuello", "Cuello ") → una sola acción (dedupe)

detectarAprendizaje — diagnósticos (APR-02):
- Dx "flacidez" en zona existente "Abdomen" que no lo tiene → diagnosticosACrear: [{ zonaNombre: "Abdomen", nombre: "Flacidez" }]
- Dx "PIEL" en "Abdomen" que ya tiene "Piel" activo → sin acciones
- Dx que matchea DiagnosticoHC inactivo de esa zona → diagnosticosAReactivar: [id]
- Dx "Otros" seleccionado → nunca se aprende
- El mismo dx existe en OTRA zona pero no en esta → SÍ se crea en esta zona (match es por zona, no global)
- Dx de una zona nueva (zona en zonasACrear) → diagnosticosACrear con zonaNombre = nombre formateado de la zona nueva

detectarAprendizaje — tratamientos (APR-03):
- Mismas reglas que diagnósticos, usando tratamientos[].nombre
- Tx nuevo en zona nueva → tratamientosACrear con zonaNombre de la zona nueva
- Dedupe dentro de la misma zona del input

General:
- otroTexto NO participa del aprendizaje (la función ni lo recibe — el input es ZonaAprendizajeInput)
- catálogo vacío + zona nueva con dx/tx → todas las acciones de creación correspondientes
