```typescript
import { Test, TestingModule } from '@nestjs/testing';
import { ConfigService } from '@nestjs/config';
import { PrismaService } from '../../prisma/prisma.service';
import { CertExpiryScheduler } from './cert-expiry.scheduler';

jest.mock('nodemailer');
const nodemailer = require('nodemailer');

describe('CertExpiryScheduler', () => {
  let scheduler: CertExpiryScheduler;
  let prisma: jest.Mocked<PrismaService>;
  let sendMailMock: jest.Mock;

  const makeConfig = (daysLeft: number) => {
    const now = new Date();
    const certExpiresAt = new Date(now.getTime() + daysLeft * 24 * 60 * 60 * 1000);
    return {
      cuit: '20123456789',
      ambiente: 'HOMOLOGACION',
      certExpiresAt,
      profesional: {
        usuario: { email: 'admin@clinic.com' },
        configClinica: { smtpHost: 'smtp.example.com', smtpPort: 587, smtpUser: 'user', smtpFrom: 'from@example.com' },
      },
    };
  };

  beforeEach(async () => {
    sendMailMock = jest.fn().mockResolvedValue({});
    nodemailer.createTransport.mockReturnValue({ sendMail: sendMailMock });

    const module: TestingModule = await Test.createTestingModule({
      providers: [
        CertExpiryScheduler,
        {
          provide: PrismaService,
          useValue: { configuracionAFIP: { findMany: jest.fn() } },
        },
        {
          provide: ConfigService,
          useValue: { get: jest.fn((key: string, def?: any) => {
            const map: Record<string, any> = {
              SMTP_HOST: 'smtp.example.com',
              SMTP_PORT: 587,
              SMTP_USER: 'user',
              SMTP_PASS: 'pass',
              SMTP_FROM: 'noreply@clinical.com',
            };
            return map[key] ?? def;
          })},
        },
      ],
    }).compile();

    scheduler = module.get<CertExpiryScheduler>(CertExpiryScheduler);
    prisma = module.get(PrismaService);
  });

  it('CERT-03: sends alert when daysLeft === 60', async () => {
    (prisma.configuracionAFIP.findMany as jest.Mock).mockResolvedValue([makeConfig(60)]);
    await scheduler.checkCertExpiry();
    expect(sendMailMock).toHaveBeenCalledTimes(1);
    expect(sendMailMock.mock.calls[0][0].subject).toContain('60');
  });

  it('CERT-03: sends alert when daysLeft === 30', async () => {
    (prisma.configuracionAFIP.findMany as jest.Mock).mockResolvedValue([makeConfig(30)]);
    await scheduler.checkCertExpiry();
    expect(sendMailMock).toHaveBeenCalledTimes(1);
  });

  it('CERT-03: does NOT send alert when daysLeft === 45', async () => {
    (prisma.configuracionAFIP.findMany as jest.Mock).mockResolvedValue([makeConfig(45)]);
    await scheduler.checkCertExpiry();
    expect(sendMailMock).not.toHaveBeenCalled();
  });

  it('CERT-03: sends alert (URGENTE) when daysLeft <= 5', async () => {
    (prisma.configuracionAFIP.findMany as jest.Mock).mockResolvedValue([makeConfig(3)]);
    await scheduler.checkCertExpiry();
    expect(sendMailMock).toHaveBeenCalledTimes(1);
    expect(sendMailMock.mock.calls[0][0].subject).toContain('URGENTE');
  });

  it('CERT-03: does not call sendMail when SMTP is not configured', async () => {
    (prisma.configuracionAFIP.findMany as jest.Mock).mockResolvedValue([{
      ...makeConfig(30),
      profesional: {
        usuario: { email: 'admin@clinic.com' },
        configClinica: null,  // no ConfigClinica
      },
    }]);
    // Override ConfigService to return undefined for SMTP vars
    jest.spyOn(module.get(ConfigService), 'get').mockReturnValue(undefined);
    await scheduler.checkCertExpiry();
    expect(sendMailMock).not.toHaveBeenCalled();
  });
});
```
