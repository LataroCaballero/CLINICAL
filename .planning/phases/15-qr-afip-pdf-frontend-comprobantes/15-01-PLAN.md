```typescript
import * as PDFDocument from 'pdfkit';

async generatePdfBuffer(data): Promise<Buffer> {
  return new Promise((resolve, reject) => {
    const doc = new PDFDocument({ margin: 50, size: 'A4' });
    const chunks: Buffer[] = [];
    doc.on('data', (chunk) => chunks.push(chunk));
    doc.on('end', () => resolve(Buffer.concat(chunks)));
    doc.on('error', reject);
    this.buildContent(doc, data).then(() => doc.end()).catch(reject);
  });
}
// doc.image(buffer, x, y, { width, height }) embeds PNG raster image
```

RG 5616/2024 QR payload interface (from 15-RESEARCH.md):
```typescript
export interface AfipQrPayload {
  ver: number;        // 1
  fecha: string;      // 'YYYY-MM-DD'
  cuit: number;       // CUIT del emisor (integer, no guiones)
  ptoVta: number;
  tipoCmp: number;    // 1=FactA, 6=FactB, 11=FactC
  nroCmp: number;
  importe: number;
  moneda: string;     // 'PES' for ARS, 'DOL' for USD (NOT 'ARS'/'USD')
  ctz: number;        // 1.000 for ARS invoices
  tipoDocRec: number; // 80=CUIT, 96=DNI, 99=SIN_ID
  nroDocRec: number;
  tipoCodAut: string; // 'E' = CAE
  codAut: number;     // CAE as number (14 digits)
}
```

AFIP moneda code mapping (critical pitfall — AFIP uses non-standard codes):
```
ARS → 'PES'
USD → 'DOL'
```

tipoDocRec / nroDocRec logic:
- factura.cuit non-null → tipoDocRec=80, nroDocRec=parseInt(factura.cuit)
- else (CONSUMIDOR_FINAL / no CUIT) → tipoDocRec=99, nroDocRec=0

AfipRealService currently selects cfg with: select: { cuit, ptoVta, ambiente }
Phase 15 requires cfg.cuit for the QR payload — already selected.
The factura fields (moneda, tipoCambio, total, cuit, fecha) must be available inside the transaction.
The AfipRealService already has `const { profesionalId, facturaId, condicionIVAReceptorId, fecha } = params` — fecha is available.
The factura DB record must be fetched inside $transaction (or params must carry moneda/tipoCambio/total/cuit).
Recommendation: fetch factura inside $transaction before advisory lock: `const factura = await tx.factura.findUniqueOrThrow({ where: { id: facturaId }, select: { moneda, tipoCambio, total, cuit } })`.
