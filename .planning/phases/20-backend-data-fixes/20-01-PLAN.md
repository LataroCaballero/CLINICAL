```typescript
// Validar y preparar fecha (después de resolver profesionalId, antes de la tx)
let fechaFinal: Date | undefined;
if (dto.fecha) {
  const parsed = new Date(dto.fecha);
  if (isNaN(parsed.getTime())) {
    throw new BadRequestException('Formato de fecha inválido.');
  }
  // Normalizar a medianoche local para comparar solo fecha
  const hoy = new Date();
  hoy.setHours(23, 59, 59, 999);
  if (parsed > hoy) {
    throw new BadRequestException('No se puede crear una entrada con fecha futura.');
  }
  fechaFinal = parsed;
}
```
