**Step A — Create `HCCreatorDialog.tsx`:**

Location: `frontend/src/components/patient/PatientDrawer/views/HCCreatorDialog.tsx`

```typescript
'use client';

import { Dialog, DialogContent, DialogHeader, DialogTitle } from '@/components/ui/dialog';
import { HCCreatorForm } from '@/components/live-turno/tabs/hc/HCCreatorForm';

export interface HCCreatorDialogProps {
  open: boolean;
  onOpenChange: (open: boolean) => void;
  pacienteId: string;
  profesionalId: string;
  obraSocialId?: string;
}

export function HCCreatorDialog({
  open,
  onOpenChange,
  pacienteId,
  profesionalId,
  obraSocialId,
}: HCCreatorDialogProps) {
  return (
    <Dialog open={open} onOpenChange={onOpenChange}>
      <DialogContent className="max-w-2xl max-h-[85vh] overflow-y-auto">
        <DialogHeader>
          <DialogTitle>Nueva entrada de Historia Clínica</DialogTitle>
        </DialogHeader>
        <HCCreatorForm
          pacienteId={pacienteId}
          profesionalId={profesionalId}
          obraSocialId={obraSocialId}
          showDatePicker={true}
          onSaved={() => onOpenChange(false)}
        />
      </DialogContent>
    </Dialog>
  );
}
```

No `turnoId` prop is passed — `turnoId` is `undefined` by design (HCDR-02: no active turno required).
